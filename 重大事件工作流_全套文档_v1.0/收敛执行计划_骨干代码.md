## 一、数据契约枚举 (Standard Enums)

```python
from dataclasses import dataclass, field
from typing import List, Optional, Dict
from enum import Enum
import concurrent.futures
import datetime

# ─── 核心枚举（严格对接各项附录）───

class EventState(str, Enum):
    INITIAL = "Initial"
    DEVELOPING = "Developing"
    PEAK = "Peak"
    FADING = "Fading"
    DEAD = "Dead"

class MacroConfirmation(str, Enum):
    SUPPORTIVE = "supportive"
    NEUTRAL = "neutral"
    HOSTILE = "hostile"

class SectorConfirmation(str, Enum):
    STRONG = "strong"
    MEDIUM = "medium"
    WEAK = "weak"
    NONE = "none"

class LeaderConfirmation(str, Enum):
    CONFIRMED = "confirmed"
    PARTIAL = "partial"
    UNCONFIRMED = "unconfirmed"
    FAILED = "failed"

class A1Validation(str, Enum):
    PASS = "pass"
    PARTIAL = "partial"
    FAIL = "fail"

class TradingState(str, Enum):
    OBSERVE = "observe"
    PROBE = "probe"
    TRADABLE = "tradable"
    ADDABLE = "addable"
    REDUCE = "reduce"
    AVOID = "avoid"

class TradeDecision(str, Enum):
    TRADABLE = "tradable"
    OBSERVE_ONLY = "observe_only"
    INTRADAY_ONLY = "intraday_only"
    OVERNIGHT_ALLOWED = "overnight_allowed"
    AVOID = "avoid"

class PositionTier(str, Enum):
    TEST = "test"
    LIGHT = "light"
    MEDIUM = "medium"
    NONE = "none"

class TargetBucket(str, Enum):
    ETF = "ETF"
    SECTOR = "Sector"
    LEADER = "Leader"
    FOLLOWER = "Follower"
```

```python
class ActionCard:
    # 事件层

    # 主路径层

    secondary_paths: List[str]

    target_bucket: str          # 枚举映射: ETF/Sector/Leader/Follower

    position_tier: str
```

## 三、逻辑映射系统与引擎实现

```python
class ActionableRunner:
    def __init__(self, ai, adjudicator, validator):
        self.ai = ai
        self.adjudicator = adjudicator
        self.validator = validator
        self._executor = concurrent.futures.ThreadPoolExecutor(max_workers=2)

    def run(self, news_text: str) -> ActionCard:
        event = self.ai.analyze_news(news_text)
        path = self.adjudicator.adjudicate(event)
        market = self.validator.validate(event, path)

        # 1. 动态窗口计算
        time_win = self._calc_time_window(event.event_time)

        # 2. 第一段映射：推导 Trading State (附录F)
        trading_state = self._map_trading_state(event.event_state, market)
        
        # 3. 软到硬的跨越：A1 强制拦截与决策推导
        decision, position = self._derive_hard_decision(trading_state, market.a1_market_validation, event)
        
        # 4. 评级计算
        grade = self._calc_grade(event, market, decision)

        # 5. 类型对齐与动态策略计算
        catalyst_val = self._map_catalyst_state(event.event_state)
        macro_val = self._map_macro_state(market.macro_confirmation)
        bucket_val, best_tgt = self._resolve_target(path)
        setup_val = self._derive_setup(decision, event.event_state)

        # 6. 生成标准卡片
        card = ActionCard(
            event_name=event.event_id,
            event_type=event.event_type,
            event_time=event.event_time,
            time_window=time_win,
            evidence_grade=event.evidence_grade,
            catalyst_state=catalyst_val,
            primary_path=path.primary_path,
            secondary_paths=path.secondary_paths,
            rejected_paths=path.rejected_paths,
            target_bucket=bucket_val,
            macro_state=macro_val,
            sector_confirmation=market.sector_confirmation.value,
            leader_confirmation=market.leader_confirmation.value,
            a1_market_validation=market.a1_market_validation.value,
            trading_state=trading_state.value,
            trade_grade=grade,
            trade_decision=decision.value,
            best_target=best_tgt,
            best_setup=setup_val,
            position_tier=position.value,
            execution_window="open" if decision != TradeDecision.INTRADAY_ONLY else "intraday",
            risk_switches=self._generate_risk_switches(market),
            invalidators=["Primary path logic breach", "Leader breaks below 20MA"],
            downgrade_rules=["Shift to Peak", "A1 shifts from PASS to PARTIAL"],
            one_line_verdict=f"[{grade}级][{decision.value}] 路径收敛于 {best_tgt}"
        )

        # 7. 安全的异步落盘
        self._executor.submit(self._log_replay_task, event, path, market, card)

        return card

    # ─── 具象化规则集合 ───

    def _calc_time_window(self, event_time_str: str) -> str:
        """根据事件时间计算时效窗口"""
        try:
            from datetime import datetime, timezone
            # 简化处理：假设传入 ISO8601, 转为 UTC 时间比对
            evt_time = datetime.fromisoformat(event_time_str.replace("Z", "+00:00"))
            now = datetime.now(timezone.utc)
            hours_diff = (now - evt_time).total_seconds() / 3600.0
            
            if hours_diff <= 24: return "0-24h"
            if hours_diff <= 48: return "24-48h"
            if hours_diff <= 72: return "48-72h"
            return ">72h"
        except Exception:
            return "0-24h"  # 降级兜底

    def _map_trading_state(self, e_state: EventState, mkt: MarketStatus) -> TradingState:
        """结合事件阶段与市场绝对强度，推断最终交易状态"""
        a1 = mkt.a1_market_validation
        if a1 == A1Validation.FAIL:
            return TradingState.AVOID

        if e_state == EventState.INITIAL:
            return TradingState.PROBE if a1 == A1Validation.PASS else TradingState.OBSERVE
            
        elif e_state == EventState.DEVELOPING:
            if a1 == A1Validation.PASS:
                # 强确认时激活 ADDABLE 动作
                if (mkt.macro_confirmation == MacroConfirmation.SUPPORTIVE and 
                    mkt.sector_confirmation == SectorConfirmation.STRONG and 
                    mkt.leader_confirmation == LeaderConfirmation.CONFIRMED):
                    return TradingState.ADDABLE
                return TradingState.TRADABLE
            return TradingState.PROBE
            
        elif e_state in (EventState.PEAK, EventState.FADING):
            return TradingState.REDUCE if a1 == A1Validation.PASS else TradingState.AVOID
            
        return TradingState.AVOID

    def _derive_hard_decision(self, t_state: TradingState, a1: A1Validation, event: EventObject):
        """交易状态 -> 具象化动作决策 (遵循 NO A1 -> NO TRADE 铁律)"""
        # 硬拦截优先
        if a1 == A1Validation.FAIL or t_state == TradingState.AVOID or event.event_state == EventState.DEAD:
            return TradeDecision.AVOID, PositionTier.NONE
            
        if event.evidence_grade == "C":
            return TradeDecision.OBSERVE_ONLY, PositionTier.NONE

        if t_state == TradingState.TRADABLE:
            return TradeDecision.TRADABLE, PositionTier.MEDIUM
        elif t_state == TradingState.PROBE:
            return TradeDecision.INTRADAY_ONLY, PositionTier.TEST
        elif t_state == TradingState.ADDABLE:
            return TradeDecision.OVERNIGHT_ALLOWED, PositionTier.MEDIUM
        elif t_state == TradingState.REDUCE:
            return TradeDecision.OBSERVE_ONLY, PositionTier.LIGHT
        else:
            return TradeDecision.OBSERVE_ONLY, PositionTier.NONE

    def _calc_grade(self, evt, mkt, dec: TradeDecision) -> str:
        if dec == TradeDecision.AVOID: return "D"
        if dec == TradeDecision.TRADABLE and mkt.macro_confirmation == MacroConfirmation.SUPPORTIVE: return "A"
        if dec in [TradeDecision.TRADABLE, TradeDecision.OVERNIGHT_ALLOWED]: return "B"
        return "C"

    def _map_catalyst_state(self, state: EventState) -> str:
        mapping = {
            EventState.INITIAL: "First Impulse",
            EventState.DEVELOPING: "Continuation",
            EventState.PEAK: "Exhaustion",
            EventState.FADING: "Exhaustion",
            EventState.DEAD: "Dead"
        }
        return mapping.get(state, "Dead")

    def _map_macro_state(self, macro: MacroConfirmation) -> str:
        mapping = {
            MacroConfirmation.SUPPORTIVE: "risk-on",
            MacroConfirmation.NEUTRAL: "mixed",
            MacroConfirmation.HOSTILE: "risk-off"
        }
        return mapping.get(macro, "risk-off")

    def _resolve_target(self, path: PathDecision):
        if path.target_leader: return TargetBucket.LEADER.value, path.target_leader[0]
        if path.target_etf: return TargetBucket.ETF.value, path.target_etf[0]
        if path.target_sector: return TargetBucket.SECTOR.value, path.target_sector[0]
        return TargetBucket.FOLLOWER.value, "N/A"

    def _derive_setup(self, dec: TradeDecision, state: EventState) -> str:
        if dec == TradeDecision.AVOID: return "avoid"
        if state == EventState.INITIAL: return "breakout"
        if state == EventState.DEVELOPING: return "pullback_confirm"
        return "buy_dip"

    def _generate_risk_switches(self, mkt: MarketStatus) -> List[str]:
        switches = []
        if mkt.macro_confirmation != MacroConfirmation.SUPPORTIVE:
            switches.append("Macro inversion immediately forces close")
        if mkt.leader_confirmation != LeaderConfirmation.CONFIRMED:
            switches.append("Leader breakdown below VWAP halts additions")
        return switches

    def _log_replay_task(self, event, path, market, card):
        # 独立线程池安全执行，无循环阻塞风险
        pass
```
